13/**
14 * Overview of Event.php
15 *
16 * This file defines the Event class in the orange\framework namespace.
17 * It is the framework’s event manager, responsible for registering, managing, and triggering events across the system.
18 * It follows the Singleton pattern (so only one instance is active) and implements the EventInterface.
19 * It also uses the ConfigurationTrait to handle event configuration.
20 *
21 * ⸻
22 *
23 * 1. Core Purpose
24 *  •   Acts as a publish–subscribe system: components can register listeners for named triggers, and those listeners will run when the trigger is fired.
25 *  •   Provides priority-based execution so some listeners can run before others.
26 *  •   Supports both closures (inline functions) and class–method pairs as event handlers.
27 *  •   Centralizes event handling, so code stays loosely coupled and extensible.
28 *
29 * ⸻
30 *
31 * 2. Initialization
32 *  •   The constructor is protected (Singleton enforced).
33 *  •   Accepts a configuration array of event mappings.
34 *  •   Uses mergeConfigWith() (from ConfigurationTrait) to combine defaults with provided configuration.
35 *  •   Initializes the $events store by looping through config and registering each event.
36 *  •   Supports a special disabled flag to globally turn events off.
37 *
38 * Example configuration format:
39 *
40 * return [
41 *   'before.router' => [
42 *       [\app\libraries\OutputCors::class . '::handleCrossOriginResourceSharing', Event::PRIORITY_HIGHEST],
43 *       [\app\libraries\Middleware::class . '::beforeRouter'],
44 *   ],
45 *   'after.controller' => [
46 *       [\app\libraries\Middleware::class . '::afterController'],
47 *   ],
48 * ];
49 *
50 * 3. Key Features
51 *  1.  Enable/Disable
52 *  •   disable() → stops all events from firing.
53 *  •   enable() → re-enables them.
54 *  2.  Registration
55 *  •   register($trigger, $callable, $priority) → adds a single event listener.
56 *  •   registerMultiple($events, $priority) → adds several at once.
57 *  •   Internally uses registerEvent() and registerClosureEvent() to store listeners.
58 *  3.  Triggering
59 *  •   trigger($trigger, &...$arguments) → fires all listeners for a given trigger, passing arguments by reference.
60 *  •   Listeners can stop propagation by returning false.
61 *  4.  Management
62 *  •   has($trigger) → checks if listeners exist for a trigger.
63 *  •   triggers() → lists all registered triggers.
64 *  •   unregister($eventId) → removes a specific listener.
65 *  •   unregisterAll($trigger = null) → clears all listeners, or just for one trigger.
66 *  5.  Listener Execution
67 *  •   Listeners are sorted by priority (highest first).
68 *  •   Supports closures directly, or array form [ClassName::class, 'method'] which will be wrapped into a closure that instantiates the class and calls the method.
69 *
70 * ⸻
71 *
72 * 4. Error Handling
73 *  •   Throws InvalidValue exception if a bad callable is registered.
74 *  •   Logs debug information (logMsg) at various points to aid troubleshooting.
75 *
76 * ⸻
77 *
78 * 5. Big Picture
79 *  •   Event.php provides the hook system for the Orange framework.
80 *  •   It allows developers to plug into the lifecycle (like before.router, after.output, etc.) without modifying core code.
81 *  •   By centralizing event registration and dispatch, it keeps the system flexible, extensible, and testable.
82 *
83 * Example Configuration:
84 * return [
85 *   'before.router' => [
86 *       [\app\libraries\OutputCors::class . '::handleCrossOriginResourceSharing', Event::PRIORITY_HIGHEST],
87 *       [\app\libraries\Middleware::class . '::beforeRouter'],
88 *   ],
89 *   'before.controller' => [
90 *       [\app\libraries\Middleware::class . '::beforeController'],
91 *   ],
92 *   'after.controller' => [

371  /**
372   * Register a closure event listener.
373   * This method generates a unique event ID based on the priority and a per-process
374   * sequence number, and stores the closure in the events array under the
375   * normalized trigger name.
376   *
377   * @param string $trigger
378   * @param Closure $callable
379   * @param int $priority
380   * @return int
381   */
382    protected function registerClosureEvent(string $trigger, \Closure $callable, int $priority): int
383    {
384      // Priority occupies the high-order digits so krsort() in listeners() sorts
385      // purely by array key and still ends up highest-priority-first; the low
386      // twelve digits are a strictly increasing per-process sequence number so two
387      // registrations - even at identical priority - never collide.
388      //
389      // Previously this concatenated $priority with hrtime(true) as strings and cast
390      // the result with intval(). hrtime(true) grows without bound as the process/
391      // system keeps running, and once the combined digit count passed PHP_INT_MAX
392      // (~19 digits), intval() silently clamped every subsequent ID to the same
393      // value - collapsing distinct listeners onto the same array key and silently
394      // dropping earlier ones. The sequence counter here is bounded by how many
395      // events get registered, not by wall-clock/uptime, so it can't drift into an
396      // overflow the way a raw timestamp can.
397        $eventId = ($priority * 1_000_000_000_000) + (self::$sequence++ % 1_000_000_000_000);
398
399        $this->events[$this->normalize($trigger)][$eventId] = $callable;
400
401        return $eventId;
402    }
